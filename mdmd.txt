CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    fio VARCHAR(255),
    phone VARCHAR(30),
    email VARCHAR(255),
    login VARCHAR(50) UNIQUE,
    password VARCHAR(255)
);

CREATE TABLE orders (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    dt DATETIME,
    weight VARCHAR(50),
    size VARCHAR(100),
    type VARCHAR(100),
    `from` TEXT,
    `to` TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
